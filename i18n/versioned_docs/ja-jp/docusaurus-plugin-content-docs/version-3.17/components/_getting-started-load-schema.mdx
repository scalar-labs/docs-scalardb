import Tabs from '@theme/Tabs';
import TabItem from '@theme/TabItem';
import CodeBlock from '@theme/CodeBlock';
import Admonition from '@theme/Admonition';
import Link from '@docusaurus/Link';

{/*
  Reusable Schema Loader command tabs for "Load the database schema" sections.

  Used by the ScalarDB Core (Java and Kotlin) getting-started guides, which share
  the same `database.properties` config file and the same `schema.json` schema.
  ScalarDB Cluster guides use a different schema-loader binary and are not handled
  by this component.

  The UI uses the same two-level nested-tabs design as <StorageSetupTabs>:
    - Outer tabs (groupId="storage-category") pick a category:
        "リレーショナルデータベース" | "NewSQL データベース" | "NoSQL データベース"
    - Inner tabs (groupId="databases") pick a specific storage in that category,
      listed in alphabetical order.

  Takes no props.
*/}

export const baseCommand = (extraFlags) =>
  `java -jar scalardb-schema-loader-<VERSION>.jar --config database.properties --schema-file schema.json --coordinator${extraFlags ? ' ' + extraFlags : ''}`;

export const loaderCommand = (extraFlags) => (
  <CodeBlock language="console">{baseCommand(extraFlags)}</CodeBlock>
);

export const CoordinatorNote = ({ children }) => (
  <Admonition type="note">
    <p>
      スキーマ内に <code>transaction</code> が <code>true</code> に設定されたテーブルが存在するため、<code>--coordinator</code> オプションが指定されています。スキーマの設定とロードの詳細については、<Link to="/ja-jp/schema-loader">ScalarDB Schema Loader</Link> を参照してください。
    </p>
    {children}
  </Admonition>
);

<Tabs groupId="storage-category" queryString>
